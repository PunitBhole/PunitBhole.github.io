/*
Create a class Rational Number (fractions) with the following capabilities: 
a) Create a constructor that prevents a 0 denominator in a fraction, reduces or simplifies fractions that are not in reduced form and avoids negative denominators. 
b) Overload the addition, subtraction, multiplication and division operators for this class. 
c) Overload the relational and equality operators for this class.
*/


#include<<iostream>
using namespace std;
class Rational
{
   int num, den;
public:
   Rational()
   {
      num=0; den=0;
   }
   void reduce()
   {
      if(num==den)
      {
         num=1;
         den=1;
      }
      else if(num%den==0)
      {
         num = num/den;
         den = 1;
      }
      else if(den%num==0)
      {
         den = den/num;
         num=1;
      }
      else if(num>den)
      {
         int i,t=den;
         for(i=1 ; num%t>0 ; i++)
         {
            t = den*i;
            if(num%i==0 && den%i==0)
            {
               num = num/i;
               den = den/i;
            }
            else
            {
               break;
            }
         }
      }
      else if(num<<den)
      {
         int i, t=num;
         for(i=1 ; den%t>0 ; i++)
         {
            t = num*i;
            if(num%i==0 && den%i==0)
            {
               num=num/i;
               den=den/i;
            }
            else
            {
            break;
            }
         }
      }
   }
   friend Rational operator+(Rational f1, Rational f2)
   {
      Rational sum;
      sum.den = f1.den*f2.den;
      sum.num = (f1.num*f2.den) + (f1.den*f2.num);
      sum.reduce();
      return sum;
   }
   friend Rational operator-(Rational f1, Rational f2)
   {
      Rational diff;
      diff.den = f1.den*f2.den;
      diff.num = (f1.num*f2.den) - (f2.num*f1.num);
      diff.reduce();
      return diff;
   }
   friend Rational operator*(Rational f1, Rational f2)
   {
      Rational prod;
      prod.num = f1.num*f1.den;
      prod.den = f1.den*f2.den;
      prod.reduce();
      return prod;
   }
   friend Rational operator/(Rational f1, Rational f2)
   {
      Rational q;
      q.num = f1.num*f2.den;
      q.den = f1.den*f2.num;
      q.reduce();
      return q;
   }
   friend ostream &operator<<(ostream &out , Rational &f)
   {
      out<<<f.num<<"/"<<<f.den<<<endl;
   }  
   friend istream &operator>>(istream &in , Rational &f)
   {
      char ch;
      in>>f.num>>ch>>f.den;
      cout<<<endl;

   }
};

    int a,b,c;
    public:
    
    Quadratic()

        a=0;
        b=0;

    }

        inp>>q.a;

        inp>>q.c;
    }

        cout<<<endl;

        Quadratic p1;
        p1.a=a+m.a;
        p1.b=b+m.b;
        p1.c=c+m.c;
        return p1;

    {
        int r,eval;
        cout<<"\nEnter the value for x which needs to be substituted :";
        cin>>r;
        eval=((t.a)*r*r)+((t.b)*r)+(t.c);
        cout<<"\nOn substituting x="<<<r<<"the value of the polynomial is :"<<<eval;
    }

        float x1,x2,m,n;

            cout<<"\nThe second root is :"<<<x2;

            cout<<"\nThe roots are imaginary";
        
    }
};
